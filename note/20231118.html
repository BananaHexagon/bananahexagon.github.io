数式とコードを表示できるようになってうれしいのでさっき作った円周率を求める式とコードを置いておきます

\(\pi = \displaystyle \lim_{n \to \infty} \sum_{i=0}^{n-1} \sqrt{\left(\sqrt{1 - \left(\displaystyle\frac{1}{n}\right)}-\sqrt{1 - \left(\displaystyle\frac{i+1}{n}\right)}\right)^2+\left(\displaystyle\frac{1}{n}\right)^2} \)

pub fn main(split: u32) -> f64 {
    let mut r = 0.0;
    for i in 0..split {
        r += part(
            (i as f64) / (split as f64),
            ((i + 1) as f64) / (split as f64),
        );
    }
    r * 2.0
}

fn part(a: f64, b: f64) -> f64 {
    let ah = (1.0 - a * a).sqrt();
    let bh = (1.0 - b * b).sqrt();
    let h = ah - bh;
    let w = b - a;
    (h * h + w * w).sqrt()
}
